const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); const scoreEl = document.getElementById('score'); let score = 0; let gameSpeed = 5; let isGameOver = false; let frame = 0; const gravity = 0.8; const floorY = 350; // 1. 角色物件 (新增 size 狀態) const player = { x: 100, y: floorY - 50, baseWidth: 40, baseHeight: 50, width: 40, height: 50, dy: 0, jumpForce: -15, grounded: false, isBig: false, // 瑪利歐變大狀態 draw() { ctx.fillStyle = this.isBig ? "#2ecc71" : "#555"; // 變大換個色 ctx.fillRect(this.x, this.y, this.width, this.height); // 畫眼睛 ctx.fillStyle = "white"; ctx.fillRect(this.x + (this.isBig ? 30 : 25), this.y + 10, 5, 5); }, update() { if (this.y + this.height < floorY) { this.dy += gravity; this.grounded = false; } else { this.dy = 0; this.grounded = true; this.y = floorY - this.height; } this.y += this.dy; this.draw(); }, grow() { if (!this.isBig) { this.isBig = true; this.width = 60; this.height = 80; this.y -= 30; } }, shrink() { this.isBig = false; this.width = 40; this.height = 50; } }; // 2. 遊戲元素陣列 let enemies = []; let items = []; let blocks = []; // --- 類別設計 --- // 怪物 (像 Goomba) class Enemy { constructor() { this.width = 30; this.height = 30; this.x = canvas.width; this.y = floorY - this.height; this.speed = gameSpeed + 1; } update() { this.x -= this.speed; ctx.fillStyle = "#8e44ad"; // 紫色怪物 ctx.fillRect(this.x, this.y, this.width, this.height); } } // 問號磚塊 class Block { constructor() { this.x = canvas.width; this.y = floorY - 150; this.width = 40; this.height = 40; this.used = false; } update() { this.x -= gameSpeed; ctx.fillStyle = this.used ? "#bdc3c7" : "#f39c12"; ctx.fillRect(this.x, this.y, this.width, this.height); if(!this.used) { ctx.fillStyle = "white"; ctx.font = "20px Arial"; ctx.fillText("?", this.x + 15, this.y + 25); } } } // 蘑菇道具 class Mushroom { constructor(x, y) { this.x = x; this.y = y; this.width = 30; this.height = 30; } update() { this.x -= gameSpeed - 1; ctx.fillStyle = "#e74c3c"; // 紅色蘑菇 ctx.beginPath(); ctx.arc(this.x + 15, this.y + 15, 15, 0, Math.PI * 2); ctx.fill(); } } // 3. 核心邏輯 function gameLoop() { if (isGameOver) return showGameOver(); ctx.clearRect(0, 0, canvas.width, canvas.height); // 繪製地板 ctx.fillStyle = "#714e3b"; ctx.fillRect(0, floorY, canvas.width, 50); player.update(); // 隨機生成 if (frame % 150 === 0) enemies.push(new Enemy()); if (frame % 250 === 0) blocks.push(new Block()); // 處理磚塊 blocks.forEach((b, i) => { b.update(); // 碰撞偵測: 玩家頭部撞擊磚塊底部 if (!b.used && player.dy < 0 && player.x < b.x + b.width && player.x + player.width > b.x && player.y < b.y + b.height && player.y > b.y) { b.used = true; items.push(new Mushroom(b.x, b.y - 40)); score += 50; } if (b.x + b.width < 0) blocks.splice(i, 1); }); // 處理蘑菇 items.forEach((m, i) => { m.update(); if (player.x < m.x + m.width && player.x + player.width > m.x && player.y < m.y + m.height && player.y + player.height > m.y) { player.grow(); items.splice(i, 1); } }); // 處理怪物 enemies.forEach((en, i) => { en.update(); if (player.x < en.x + en.width && player.x + player.width > en.x && player.y < en.y + en.height && player.y + player.height > en.y) { if (player.isBig) { player.shrink(); enemies.splice(i, 1); // 碰到怪物變小，怪物消失 } else { isGameOver = true; } } if (en.x + en.width < 0) { enemies.splice(i, 1); score += 10; } }); scoreEl.innerText = score; frame++; requestAnimationFrame(gameLoop); } function showGameOver() { ctx.fillStyle = "rgba(0,0,0,0.7)"; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = "white"; ctx.textAlign = "center"; ctx.font = "40px Arial"; ctx.fillText("GAME OVER", canvas.width/2, canvas.height/2); ctx.font = "20px Arial"; ctx.fillText("按下 R 重新開始", canvas.width/2, canvas.height/2 + 50); } window.addEventListener('keydown', (e) => { if ((e.code === 'Space' || e.code === 'ArrowUp') && player.grounded) { player.dy = player.jumpForce; } if (e.code === 'KeyR' && isGameOver) location.reload(); }); gameLoop();
