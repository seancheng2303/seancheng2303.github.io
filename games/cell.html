/** * 《細胞倖存者：究極進化》核心遊戲引擎 * 適用於 HTML5 Canvas 2D */ // --- 1. 資源與路徑設定 --- const ASSET_PATHS = { neutrophil: "asset/cell/cell_neutrophil.png", macrophage: "asset/cell/cell_macrophage.png", killer_t: "asset/cell/cell_killer_t.png", adeno: "asset/cell/virus_adeno.png", noro: "asset/cell/virus_noro.png", phage: "asset/cell/virus_phage.png", amoeba: "asset/cell/microbe_amoeba.png", corona: "asset/cell/virus_corona.png" }; const IMAGES = {}; // --- 2. 角色與怪物設定資料庫 (來自規格書) --- const CHARACTER_DB = { neutrophil: { name: "🎯 嗜中性球", r: 18, speed: 3.4, maxHp: 100, weapon: "活性氧彈", talent: "細胞吞噬" }, macrophage: { name: "🛡️ 巨噬細胞", r: 23, speed: 2.9, maxHp: 150, weapon: "偽足掃擊", talent: "趨化引力" }, killer_t: { name: "⚡ 殺手 T 細胞", r: 18, speed: 4.1, maxHp: 100, weapon: "穿孔素穿刺", talent: "凋亡指令" } }; const ENEMY_DB = [ { type: "adeno", name: "🟢 腺病毒", time: 0, hp: 1, dmg: 10, speed: 1.3, r: 8 }, { type: "noro", name: "🔴 諾羅病毒", time: 20, hp: 1, dmg: 15, speed: 2.4, r: 10 }, { type: "phage", name: "🟠 噬菌體", time: 55, hp: 3, dmg: 22, speed: 0.9, r: 14 }, { type: "amoeba", name: "🔮 變形蟲", time: 90, hp: 7, dmg: 35, speed: 0.6, r: 20 }, { type: "corona", name: "👑 冠狀病毒王", time: 120, hp: 80, dmg: 45, speed: 1.1, r: 40 } ]; // --- 3. 遊戲主狀態物件 --- const Game = { canvas: null, ctx: null, isPaused: false, gameTime: 0, // 秒 killCount: 0, // 玩家狀態 player: { x: 0, y: 0, r: 18, speed: 3.4, hp: 100, maxHp: 100, xp: 0, xpNeeded: 100, level: 1, id: 'neutrophil', skills: [] }, // 鏡頭狀態 camera: { x: 0, y: 0 }, // 虛擬搖桿狀態 joystick: { baseX: 0, baseY: 0, stickX: 0, stickY: 0, maxRadius: 50, isDragging: false, vx: 0, vy: 0 }, // 遊戲內實體陣列 enemies: [], bullets: [], crystals: [], // 經驗值晶體 // 初始化遊戲 init(canvasId) { this.canvas = document.getElementById(canvasId); this.ctx = this.canvas.getContext("2d"); this.resizeCanvas(); window.addEventListener('resize', () => this.resizeCanvas()); // 綁定控制事件 this.bindJoystickEvents(); // 預載入圖片 this.preload(() => { console.log("遊戲資源載入完成，等待 selectCharacter() 觸發。"); }); }, resizeCanvas() { this.canvas.width = window.innerWidth; this.canvas.height = window.innerHeight; // 初始化搖桿位置到右下角 this.joystick.baseX = this.canvas.width - 100; this.joystick.baseY = this.canvas.height - 100; this.joystick.stickX = this.joystick.baseX; this.joystick.stickY = this.joystick.baseY; }, preload(callback) { let loaded = 0; const total = Object.keys(ASSET_PATHS).length; for (let key in ASSET_PATHS) { IMAGES[key] = new Image(); IMAGES[key].src = ASSET_PATHS[key]; IMAGES[key].onload = () => { loaded++; if (loaded === total) callback(); }; } } }; // --- 4. 角色選擇與生命週期流程 --- function selectCharacter(charId) { const data = CHARACTER_DB[charId]; if (!data) return; // 初始化玩家資料 Game.player = { x: 0, y: 0, id: charId, name: data.name, r: data.r, speed: data.speed, hp: data.maxHp, maxHp: data.maxHp, xp: 0, xpNeeded: 100, level: 1, skills: [data.weapon] // 初始解鎖主武器 }; Game.isPaused = false; Game.gameTime = 0; Game.killCount = 0; Game.enemies = []; Game.bullets = []; Game.crystals = []; // 啟動計時器 (每秒+1秒) if (Game.timeInterval) clearInterval(Game.timeInterval); Game.timeInterval = setInterval(() => { if (!Game.isPaused) Game.gameTime++; }, 1000); // 開啟 gameLoop requestAnimationFrame(gameLoop); } // --- 5. 虛擬搖桿運算邏輯 --- Game.bindJoystickEvents = function() { const handleStart = (ex, ey) => { // 如果點擊範圍在右下角搖桿區域 const dist = Math.hypot(ex - this.joystick.baseX, ey - this.joystick.baseY); if (dist < this.joystick.maxRadius * 2) { this.joystick.isDragging = true; handleMove(ex, ey); } }; const handleMove = (ex, ey) => { if (!this.joystick.isDragging) return; const dx = ex - this.joystick.baseX; const dy = ey - this.joystick.baseY; const dist = Math.hypot(dx, dy); const angle = Math.atan2(dy, dx); // 規格書：當 Dist > 最大半徑時，搖桿頭固定於最大半徑邊緣 if (dist > this.joystick.maxRadius) { this.joystick.stickX = this.joystick.baseX + Math.cos(angle) * this.joystick.maxRadius; this.joystick.stickY = this.joystick.baseY + Math.sin(angle) * this.joystick.maxRadius; } else { this.joystick.stickX = ex; this.joystick.stickY = ey; } // 規格書：輸出歸一化的速度向量 [-1, 1] this.joystick.vx = Math.cos(angle) * Math.min(1, dist / this.joystick.maxRadius); this.joystick.vy = Math.sin(angle) * Math.min(1, dist / this.joystick.maxRadius); }; const handleEnd = () => { this.joystick.isDragging = false; this.joystick.stickX = this.joystick.baseX; this.joystick.stickY = this.joystick.baseY; this.joystick.vx = 0; this.joystick.vy = 0; }; // 監聽觸控與滑鼠 window.addEventListener('mousedown', e => handleStart(e.clientX, e.clientY)); window.addEventListener('mousemove', e => handleMove(e.clientX, e.clientY)); window.addEventListener('mouseup', () => handleEnd()); window.addEventListener('touchstart', e => { const t = e.touches[0]; handleStart(t.clientX, t.clientY); }); window.addEventListener('touchmove', e => { const t = e.touches[0]; handleMove(t.clientX, t.clientY); }); window.addEventListener('touchend', () => handleEnd()); }; // --- 6. 核心更新邏輯 (Update) --- function update() { if (Game.isPaused) return; const p = Game.player; // 1. 玩家位移 p.x += Game.joystick.vx * p.speed; p.y += Game.joystick.vy * p.speed; // 2. 鏡頭平滑跟隨玩家（玩家保持在螢幕幾何中心） Game.camera.x = p.x; Game.camera.y = p.y; // 3. 病毒 AI 更新（追擊玩家） Game.enemies.forEach(enemy => { const dx = p.x - enemy.x; const dy = p.y - enemy.y; const dist = Math.hypot(dx, dy); // 基礎移動 AI if (dist > 0) { enemy.x += (dx / dist) * enemy.speed; enemy.y += (dy / dist) * enemy.speed; } // 碰撞檢查：傷害玩家 if (dist < p.r + enemy.r) { p.hp -= enemy.dmg / 60; // 假設 60 FPS 持續扣血 if (p.hp <= 0) gameOver(); } }); // 4. 經驗值晶體磁鐵吸引 Game.crystals.forEach((cry, idx) => { const dx = p.x - cry.x; const dy = p.y - cry.y; const dist = Math.hypot(dx, dy); // 巨噬細胞基礎吸收範圍增加 60% const magnetRadius = (p.id === 'macrophage') ? 160 : 100; if (dist < magnetRadius) { cry.x += (dx / dist) * 5; cry.y += (dy / dist) * 5; } // 吃到經驗球 if (dist < p.r + 5) { let gainXp = 10; // 嗜中性球 15% 機率雙倍經驗 if (p.id === 'neutrophil' && Math.random() < 0.15) gainXp *= 2; p.xp += gainXp; Game.crystals.splice(idx, 1); // 升級檢測 if (p.xp >= p.xpNeeded) { triggerUpgrade(); } } }); // 生怪邏輯模擬 (依據生存時間解鎖新怪物) if (Math.random() < 0.02) { // 隨機頻率生成 spawnEnemy(); } } // --- 7. 核心渲染邏輯 (Render) --- function render() { const ctx = Game.ctx; const canvas = Game.canvas; const p = Game.player; const cam = Game.camera; ctx.clearRect(0, 0, canvas.width, canvas.height); // 1. 繪製無限網格背景 (規格書：用數學取模運算 camera.x % 60) ctx.save(); ctx.strokeStyle = "#1a2636"; ctx.lineWidth = 1; const offsetX = -(cam.x % 60); const offsetY = -(cam.y % 60); for (let x = offsetX; x < canvas.width; x += 60) { ctx.beginPath(); ctx.moveTo(x, 0); ctx.lineTo(x, canvas.height); ctx.stroke(); } for (let y = offsetY; y < canvas.height; y += 60) { ctx.beginPath(); ctx.moveTo(0, y); ctx.lineTo(canvas.width, y); ctx.stroke(); } ctx.restore(); // 計算螢幕中心點 const cx = canvas.width / 2; const cy = canvas.height / 2; // 2. 繪製經驗值晶體 Game.crystals.forEach(cry => { const rx = cry.x - cam.x + cx; const ry = cry.y - cam.y + cy; ctx.fillStyle = "#00ffcc"; ctx.fillRect(rx - 3, ry - 3, 6, 6); }); // 3. 繪製敵人 (規格書公式：物件座標 - Camera座標 + 螢幕中心偏移) Game.enemies.forEach(enemy => { const rx = enemy.x - cam.x + cx; const ry = enemy.y - cam.y + cy; const size = enemy.r * 2; const img = IMAGES[enemy.type]; if (img) { ctx.drawImage(img, rx - enemy.r, ry - enemy.r, size, size); } else { // 後備幾何圖形 (防止圖片載入失敗) ctx.beginPath(); ctx.arc(rx, ry, enemy.r, 0, Math.PI*2); ctx.fillStyle = "red"; ctx.fill(); } }); // 4. 繪製玩家角色 (永遠在螢幕正中央) const pSize = p.r * 2; const pImg = IMAGES[p.id]; if (pImg) { ctx.drawImage(pImg, cx - p.r, cy - p.r, pSize, pSize); } else { ctx.beginPath(); ctx.arc(cx, cy, p.r, 0, Math.PI*2); ctx.fillStyle = "blue"; ctx.fill(); } // 5. 繪製虛擬搖桿 UI if (Game.joystick.isDragging || true) { // 搖桿底座 ctx.beginPath(); ctx.arc(Game.joystick.baseX, Game.joystick.baseY, Game.joystick.maxRadius, 0, Math.PI*2); ctx.fillStyle = "rgba(255, 255, 255, 0.1)"; ctx.fill(); ctx.strokeStyle = "rgba(255, 255, 255, 0.3)"; ctx.stroke(); // 搖桿頭 ctx.beginPath(); ctx.arc(Game.joystick.stickX, Game.joystick.stickY, 20, 0, Math.PI*2); ctx.fillStyle = "rgba(255, 255, 255, 0.5)"; ctx.fill(); } // 6. 繪製頂部 UI (時間、血條、經驗條) drawUI(ctx, canvas, p); } // --- 8. 輔助機制 (生怪、升級、結束) --- function spawnEnemy() { // 根據當前遊戲時間篩選可生成的病毒 const available = ENEMY_DB.filter(e => Game.gameTime >= e.time); if (available.length === 0) return; const template = available[Math.floor(Math.random() * available.length)]; // 在螢幕外隨機半徑生成 const angle = Math.random() * Math.PI * 2; const spawnDist = Math.max(window.innerWidth, window.innerHeight) / 2 + 50; Game.enemies.push({ ...template, x: Game.player.x + Math.cos(angle) * spawnDist, y: Game.player.y + Math.sin(angle) * spawnDist, hp: template.hp }); } function drawUI(ctx, canvas, p) { // 時間顯示 ctx.fillStyle = "#fff"; ctx.font = "20px Arial"; ctx.textAlign = "center"; const min = String(Math.floor(Game.gameTime / 60)).padStart(2, '0'); const sec = String(Game.gameTime % 60).padStart(2, '0'); ctx.fillText(`${min}:${sec}`, canvas.width / 2, 40); // 擊殺數 ctx.textAlign = "left"; ctx.fillText(`🧬 擊殺: ${Game.killCount}`, 20, 40); // 生命值條 ctx.fillStyle = "rgba(0,0,0,0.5)"; ctx.fillRect(20, 60, 200, 15); ctx.fillStyle = "#ff3333"; ctx.fillRect(20, 60, (Math.max(0, p.hp) / p.maxHp) * 200, 15); // 經驗值條 ctx.fillStyle = "rgba(0,0,0,0.5)"; ctx.fillRect(20, 85, 200, 8); ctx.fillStyle = "#00ffcc"; ctx.fillRect(20, 85, (p.xp / p.xpNeeded) * 200, 8); } function triggerUpgrade() { Game.isPaused = true; Game.player.xp = 0; Game.player.level++; Game.player.xpNeeded = Math.floor(Game.player.xpNeeded * 1.3); // 規格書：彈出隨機三選一突變選單 console.log("=== 🧬 細胞發生突變！請選擇升級 ==="); // 在實際網頁中，此處應顯示 HTML 彈出視窗，點擊後重設 Game.isPaused = false setTimeout(() => { Game.isPaused = false; }, 1000); // 模擬選擇後繼續 } function gameOver() { Game.isPaused = true; clearInterval(Game.timeInterval); alert(`【遊戲結束】\n生存時間：${Game.gameTime} 秒\n擊殺數：${Game.killCount}`); location.reload(); // 規格書：網頁自動刷新重置 } // 主循環 function gameLoop() { update(); render(); requestAnimationFrame(gameLoop); }
